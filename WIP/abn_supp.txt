airborne_support_artillery_upgrade = {
	
	allow = {
		or = {
			and = {
				has_country_flag = AIRBORNE_ARTILLERY_UPGRADES
				artillery_activation = 3
				not = { airborne_support_artillery_upgrade = 1 }
			}
			and = {
				has_country_flag = AIRBORNE_ARTILLERY_UPGRADES
				artillery_activation = 4
				not = { airborne_support_artillery_upgrade = 2 }
			}
			and = {
				has_country_flag = AIRBORNE_ARTILLERY_UPGRADES
				artillery_activation = 5
				not = { airborne_support_artillery_upgrade = 3 }
			}
			and = {
				has_country_flag = AIRBORNE_ARTILLERY_UPGRADES
				artillery_activation = 6
				not = { airborne_support_artillery_upgrade = 4 }
			}
			and = {
				has_country_flag = AIRBORNE_ARTILLERY_UPGRADES
				artillery_activation = 7
				not = { airborne_support_artillery_upgrade = 5 }
			}
			and = {
				has_country_flag = AIRBORNE_ARTILLERY_UPGRADES
				artillery_activation = 8
				not = { airborne_support_artillery_upgrade = 6 }
			}
		}
	}
	
	airborne_combined_arms_brigade = {}
	
	
	on_completion = artillery_theory

	difficulty = 1
	
	research_bonus_from = {
		artillery_practical = 2
		artillery_theory = 2
	}

	# common for all techs.
	start_year = 1936
	first_offset = 1938		# 2nd model is from 1938
	additional_offset = 2	# one new every 2 years
	max_level = 6
	folder = special_forces_doctrine_folder
}
