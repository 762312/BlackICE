
# Start of Santa Clausewitz' journey
country_event = {

	id = 999988

	trigger = {
		tag = OMG
		ai = yes
		month = 11 # Dec
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = { factor = 99999999  OMG = { has_country_flag = santa_is_here } }
	}
	
	title = "You Suddenly Feel Very Cheerful!"
	desc = "While civilizations squabble over territory and resources, a sudden feeling of calm, then cheerfulness, overcomes you. This is truly a magical time of year!"
	picture = "santa-clausewitz"
	
	option = {
		name = "OK"

		OMG = { set_country_flag = santa_is_here }
		
		# Make sure Santa has a base to fly back to on the OMG island, otherwise, he'll just sit there!
		14179 = { air_base = 1 }
		
		random_list = {
			16 = { load_oob = santa_clausewitz1.txt } # Thule/Greenland - closest thing we have to the north pole!
			16 = { load_oob = santa_clausewitz2.txt } # Paris
			17 = { load_oob = santa_clausewitz3.txt } # Washington, DC
			17 = { load_oob = santa_clausewitz4.txt } # Tokyo
			17 = { load_oob = santa_clausewitz5.txt } # Moscow
			17 = { load_oob = santa_clausewitz6.txt } # Berlin
		}
		
		ai_chance = { factor = 100 }
	}
}
