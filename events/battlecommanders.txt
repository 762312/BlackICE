	id = 14731
	
	trigger = {
		tag = GER
		has_country_flag = new_battlecommander
		not = { has_country_flag = new_commander_I }
	}
	
	title = "A new Battle Commander has entered the battlefield!"
	desc = "From the cold ashes of endless fights the nemesis of our enemy is born."
	picture = "bcManstein"
	
	option = {
		name = "Forward to Victory"
		#kill_leader = 310
		add_country_modifier = {
			name = "battlecommanderpenalty"
			duration = 30
		}
		load_oob = "battle_commander_GER_I.txt" 
		clr_country_flag = new_battlecommander
		set_country_flag = new_commander_I
	} 
	option = {
		name = "We have a better use for our officers"
		clr_country_flag = new_battlecommander
		set_country_flag = new_commander_I
	}
}
