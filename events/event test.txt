			controller = {
				grand_battle_plan = 1 
				war_with = THIS
			}
		}

		not = { last_battle_loser_losses = 2000 }
	}

					country_units_in_province = THIS

						limit = { 
							controller = { 
								grand_battle_plan = 1 
								war_with = THIS

									blitzkrieg = 1
									superior_firepower = 1
									deep_battle_doctrine = 1
								}
							} 
							units_in_province = 1 
						}
					}
					set_province_flag = learn_grand_battle_plan_province_flag_2

			}

country_event = { 
	id = 969760
	
	trigger = {
		not = { has_country_modifier = buy_resource_chromite_ENG }
		or = {
			has_country_modifier = buy_resource_chromite_ENG_cooldown_I
			has_country_modifier = buy_resource_chromite_ENG_cooldown_II
		}
	}

	mean_time_to_happen = {
		days = 1
	}
	
	title = "Trade Expired"
	desc = "Our trade deal for Chromite with United Kingdom has expired. We need to refresh it or find different source."
	picture = "chromium_trade_expired"

	option = {
		name = "Time to browse our options again."
		ENG = { change_variable = { which = "trade_resource_chromite_ENG" value = -1 } }	
		remove_country_modifier = buy_resource_chromite_ENG_cooldown_I
		remove_country_modifier = buy_resource_chromite_ENG_cooldown_II		
	}
}
